# Stack

Flexbox-based layout components for arranging elements in vertical (VStack) or horizontal (HStack) directions.

## API

```tsx
type Props<T extends keyof React.JSX.IntrinsicElements> =
  React.PropsWithChildren<{
    inline?: boolean;
    gap?: number | string;
    align?: "start" | "center" | "end" | "stretch" | "baseline";
    justify?:
      | "start"
      | "center"
      | "end"
      | "space-between"
      | "space-around"
      | "space-evenly";
    padding?: number | string;
    as?: T;
  }> &
    React.ComponentProps<"div">;

export const VStack = <T extends keyof React.JSX.IntrinsicElements>({
  inline,
  children,
  gap = 0,
  align = "stretch",
  justify = "start",
  padding,
  style = {},
  as = "div" as T,
  ...props
}: Props<T>) => React.ReactNode;

export const HStack = <T extends keyof React.JSX.IntrinsicElements>({
  inline,
  children,
  gap = 0,
  align = "stretch",
  justify = "start",
  padding,
  style = {},
  as = "div" as T,
  ...props
}: Props<T>) => React.ReactNode;
```

## Example

<StackExample />

```tsx
import { VStack, HStack } from "@fewings/react/components";

function LayoutExample() {
  return (
    <div>
      {/* Vertical Stack */}
      <VStack
        gap={16}
        align="center"
        justify="space-between"
        style={{ border: "1px solid #ccc", padding: "1rem", height: "300px" }}
      >
        <div>Item 1</div>
        <div>Item 2</div>
        <div>Item 3</div>
      </VStack>

      {/* Horizontal Stack */}
      <HStack
        gap={16}
        align="center"
        justify="space-between"
        style={{ border: "1px solid #ccc", padding: "1rem", marginTop: "1rem" }}
      >
        <div>Item 1</div>
        <div>Item 2</div>
        <div>Item 3</div>
      </HStack>
    </div>
  );
}
```
