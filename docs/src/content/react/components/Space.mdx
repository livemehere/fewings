# Space

A utility component for adding fixed dimensions of space in layouts. It can be used to create horizontal or vertical spacing between elements.

## API

```tsx
export const Space = ({ x, y }: { x?: string | number; y?: string | number }) =>
  React.ReactNode;
```

## Example

```tsx
import { Space } from "@fewings/react/components";

function SpacingExample() {
  return (
    <div>
      {/* Vertical spacing */}
      <div>First element</div>
      <Space y={20} />
      <div>Second element</div>
      <Space y={20} />
      <div>Third element</div>

      {/* Horizontal spacing */}
      <div style={{ display: "flex", alignItems: "center" }}>
        <div>First</div>
        <Space x={40} />
        <div>Second</div>
        <Space x={40} />
        <div>Third</div>
      </div>
    </div>
  );
}
```
