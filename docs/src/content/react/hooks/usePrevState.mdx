# usePrevState

A hook that returns the previous state value. It uses `useRef` to store the previous value and updates it after each render.

## API

```tsx
declare const usePrevState: <T>(state: T | undefined) => T | undefined;
```

## Example

```tsx
import { usePrevState } from '@fewings/react/hooks';
import { useState } from 'react';

function MyComponent() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevState(count);

  return (
    <div>
      <p>Current count: {count}</p>
      <p>Previous count: {prevCount ?? 'No previous value'}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}
```
