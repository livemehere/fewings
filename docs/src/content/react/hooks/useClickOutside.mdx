# useClickOutside

A hook that detects clicks outside of a specified element. It can also exclude certain elements from triggering the outside click handler.

## API

```tsx
declare const useClickOutside: (
  handler: (event: MouseEvent) => void,
  exclude?: RefObject<HTMLElement | null>
) => RefObject<HTMLElement | null>;
```

## Example

```tsx
import { useClickOutside } from '@fewings/react/hooks';

function Dropdown() {
  const [isOpen, setIsOpen] = useState(false);
  const dropdownRef = useClickOutside(() => setIsOpen(false));
  const excludeRef = useRef<HTMLUListElement>(null); // prevent popover element's click event from triggering the outside click handler

  return (
    <div ref={dropdownRef}>
      <button onClick={() => setIsOpen(!isOpen)}>Toggle Dropdown</button>
      {isOpen && (
        <ul ref={excludeRef}>
          <li>Item 1</li>
          <li>Item 2</li>
          <li>Item 3</li>
        </ul>
      )}
    </div>
  );
}
```
