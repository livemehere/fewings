# useMount

A hook that checks if a component has been mounted. It returns `true` after the component's initial render.

## API

```tsx
declare const useMount: () => boolean;
```

## Example

```tsx
import { useMount } from "@fewings/react/hooks";
import { useEffect } from "react";

function MyComponent() {
  const isMounted = useMount();

  useEffect(() => {
    if (isMounted) {
      console.log("Component is now mounted");
    }
  }, [isMounted]);

  return (
    <div>
      {isMounted ? "Component is mounted" : "Component is not mounted yet"}
    </div>
  );
}
```
