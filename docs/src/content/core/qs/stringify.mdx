# stringify

> Convert an object to a query string.

- `undefined` and `null` will be ignored
- if value is an array, it will be joined by `,` and vice versa

## API

```ts
type QsValue = string | string[] | undefined | null;
declare function stringify(
  obj: Record<string, QsValue>,
  prefix?: string
): string;
```

## Example

```ts
import { stringify } from '@fewings/core/qs';

console.log(
  stringify({
    name: 'kong',
    idx: ['3', '4'],
    serviceId: '430011481',
    serverId: '5',
    placementId: null,
    empty: undefined,
  })
); // ?name=kong&idx=3%2C4&serviceId=430011481&serverId=5
```
