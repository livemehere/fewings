# getCommonPath

> return the common path between two paths

## API

```ts
declare function getCommonPath(path1: string, path2: string): string;
```

## Example

```ts
import { getCommonPath } from "@fewings/core/path";

console.log(getCommonPath("/path/to/file1.txt", "/path/to/file2.txt")); // /path/to
console.log(getCommonPath("/path/to/other/file1.txt", "/path/to/file2.txt")); // /path/to
```
