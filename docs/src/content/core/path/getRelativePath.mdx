# getRelativePath

> return the relative path from one path to another

## API

```ts
declare function getRelativePath(from: string, to: string): string;
```

## Example

```ts
import { getRelativePath } from '@fewings/core/path';

console.log(getRelativePath('/path/to/file1.txt', '/path/to/file2.txt')); // file2.txt
console.log(getRelativePath('/path/to/file1.txt', '/path/to/other/file2.txt')); // ../other/file2.txt
```
